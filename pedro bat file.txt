export NG_MODELS=/ng/models

export NG_OUTPUT=/ng/output

# Class of RDF generator. 
# modelfilepath - where the model files are
# outputfile - where the finished files will be. 
# logging the output of Karma

# Loads the data. Applys R2RML model. Publish RDF. 
# Do this every time that you make changes to the database. 
# BENCHMARK - 1 hour. 
time mvn exec:java -Dexec.mainClass="edu.isi.karma.rdf.OfflineRdfGenerator" -Dexec.args="--sourcetype DB --modelfilepath $NG_MODELS/gmi/eqp/eqp/model.ttl --outputfile $NG_OUTPUT/eqp.n3 --dbtype MySQL --hostname sigma.isi.edu --username karma --password ontology --portnumber 3306 --dbname gmi --tablename eqp" &> $NG_OUTPUT/logs/eqp.log
time mvn exec:java -Dexec.mainClass="edu.isi.karma.rdf.OfflineRdfGenerator" -Dexec.args="--sourcetype DB --modelfilepath $NG_MODELS/gmi/eqp/vw_eqp_tie/model.ttl --outputfile $NG_OUTPUT/vw_eqp_tie.n3 --dbtype MySQL --hostname sigma.isi.edu --username karma --password ontology --portnumber 3306 --dbname gmi --tablename vw_eqp_tie" &> $NG_OUTPUT/logs/vw_eqp_tie.log
time mvn exec:java -Dexec.mainClass="edu.isi.karma.rdf.OfflineRdfGenerator" -Dexec.args="--sourcetype DB --modelfilepath $NG_MODELS/gmi/fac/vw_fac_tie/model.ttl --outputfile $NG_OUTPUT/vw_fac_tie.n3 --dbtype MySQL --hostname sigma.isi.edu --username karma --password ontology --portnumber 3306 --dbname gmi --tablename vw_fac_tie" &> $NG_OUTPUT/logs/vw_fac_tie.log
